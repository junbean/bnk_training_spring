첫번째 - 회원과 공연 예매 시스템 구현
	기본 기능
		1. 사용자에게 회원가입과 로그인 기능을 제공
		2. 메인페이지에서 공연 목록을 제공
		3. 목록에서 공연을 선택 시 공연 상세 페이지 이동 후 공연의 상세 정보 및 예매버튼 제공
		4. 사용자는 공연을 선택하여 예매기능 구현

			CREATE TABLE MEMBER (
			    MEMBER_ID      VARCHAR2(50) PRIMARY KEY,
			    PASSWORD       VARCHAR2(100) NOT NULL,
			    NAME           VARCHAR2(100) NOT NULL,
			    EMAIL          VARCHAR2(100),
			    PHONE          VARCHAR2(20),
			    ROLE           VARCHAR2(20) DEFAULT 'USER' CHECK (ROLE IN ('USER', 'ADMIN')), -- ← 여기 추가
			    JOIN_DATE      DATE DEFAULT SYSDATE
			);

			CREATE TABLE RESERVATION (
			    RESERVATION_ID  NUMBER PRIMARY KEY,
			    MEMBER_ID       VARCHAR2(50) REFERENCES MEMBER(MEMBER_ID),
